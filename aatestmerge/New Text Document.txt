public function redirectTo(){

    // User role
    $role = Auth::user()->role->name;

    // Check user role
    switch ($role) {
        case 'Manager':
                return '/dashboard';
            break;
        case 'Employee':
                return '/projects';
            break;
        default:
                return '/login';
            break;
    }
}
